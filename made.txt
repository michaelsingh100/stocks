	if(document.cookie.length<=0)
    	return "";

   	name += "=";
    var start = document.cookie.indexOf(name);
    if (start < 0)
       return "";

	start += name.length;

    if (end < 0)
    	end = document.cookie.length;
              
   	return document.cookie.substring(start,end);
